data <- data[1:20,]
object <- "object_7"


send_data <- function(row) {
  # Change to JSON
  data_json <- toJSON(as.list(row), auto_unbox = TRUE)
  # Post data to knack
  api_url <-
    paste0("https://api.knack.com/v1/objects/", object, "/records")
  future_promise(
    POST(
      api_url,
      add_headers(
        "X-Knack-Application-Id" = getOption("api_id"),
        "X-Knack-REST-API-Key" = getOption("api_key"),
        "Content-Type" = "application/json"
      ),
      body = data_json
    )
  )
  
}

status <- promise_all(promise_map(data.frame(t(data)), send_data))
